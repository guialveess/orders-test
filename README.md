# Git Commands for Initial Commit

Copy and paste the following commands in your terminal:

```bash
git add .dockerignore
git commit -m "chore: add dockerignore file"

git add .env.test
git commit -m "chore: add test environment file"

git add .gitignore
git commit -m "chore: add gitignore file"

git add Dockerfile
git commit -m "ci: add dockerfile"

git add README.md
git commit -m "docs: add readme file"

git add app.ts
git commit -m "feat: add app entry point"

git add bun.lock
git commit -m "chore: add bun lock file"

git add docker-compose.yml
git commit -m "ci: add docker compose configuration"

git add package.json
git commit -m "chore: add package manifest"

git add server.ts
git commit -m "feat: add server entry point"

git add tsconfig.json
git commit -m "chore: add typescript configuration"

git add vitest.config.ts
git commit -m "test: add vitest configuration"

git add config/database.ts
git commit -m "config: add database configuration"

git add config/swagger.ts
git commit -m "config: add swagger configuration"

git add middlewares/auth.middleware.ts
git commit -m "feat: add authentication middleware"

git add routes/auth.routes.ts
git commit -m "feat: add authentication routes"

git add routes/index.ts
git commit -m "feat: add routes index"

git add routes/orders.routes.ts
git commit -m "feat: add orders routes"

git add src/@types/express/index.d.ts
git commit -m "types: add express type definitions"

git add src/models/orders/order.enums.ts
git commit -m "feat: add order enums"

git add src/models/orders/order.interface.ts
git commit -m "feat: add order interface"

git add src/models/orders/order.schema.ts
git commit -m "feat: add order schema"

git add src/models/users/user.interface.ts
git commit -m "feat: add user interface"

git add src/models/users/user.schema.ts
git commit -m "feat: add user schema"

git add test/app.ts
git commit -m "test: add app test file"

git add test/auth.routes.test.ts
git commit -m "test: add auth routes tests"

git add test/orders.routes.test.ts
git commit -m "test: add orders routes tests"

git add test/setup.ts
git commit -m "test: add test setup file"

git add test/utils.ts
git commit -m "test: add test utilities"
```
